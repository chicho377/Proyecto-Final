Proceso VentaEntradasTeatro
	
    // Inicializar variables
    numeroFactura <- 0
    cantGaleria <- 0
    cantPalco <- 0
    cantTotal <- 0
    cedula <- 0
    tipoCliente <- 0
    tipoTiquete <- 0
    cantidadTiquetes <- 0
    montoUnitario <- 0.0
    montoVenta <- 0.0
    descuento <- 0.0
    subTotal <- 0.0
    impuesto <- 0.0
    totalPagar <- 0.0
    acumGaleria <- 0.0
    acumPalco <- 0.0
    acumTotal <- 0.0
    nombre <- ""
    continuar <- 's'
	
    Mientras continuar = 's' O continuar = 'S' Hacer
        // Incrementar el número de factura
        numeroFactura <- numeroFactura + 1
		
        // Ingresar datos del cliente
        Escribir "Ingrese cedula: "
        Leer cedula
        Escribir "Ingrese nombre y apellido: "
        Leer nombre
		
        Escribir "Ingrese tipo de cliente (1=Nino o adulto mayor, 2=Adulto): "
        Leer tipoCliente
		
        // Validar que el tipo de cliente sea correcto
        Mientras tipoCliente > 2 O tipoCliente < 1 Hacer
            Escribir "Tipo de cliente incorrecto. Por favor ingrese 1-Nino o adulto mayor, 2-Adulto: "
            Leer tipoCliente
        FinMientras
		
        Escribir "Ingrese tipo de tiquete (1=Tiquete Galeria, 2=Tiquete Palco): "
        Leer tipoTiquete
		
        // Validar que el tipo de tiquete sea correcto
        Mientras tipoTiquete > 2 O tipoTiquete < 1 Hacer
            Escribir "Tipo de tiquete incorrecto. Por favor ingrese 1-Galeria, 2-Palco: "
            Leer tipoTiquete
        FinMientras
		
        Escribir "Ingrese cantidad de tiquetes: "
        Leer cantidadTiquetes
		
        // Calcular monto unitario
        Si tipoTiquete = 1 Entonces
            montoUnitario <- 6000
        Sino
            montoUnitario <- 12000
        FinSi
		
        // Calcular monto de venta
        montoVenta <- cantidadTiquetes * montoUnitario
		
        // Calcular descuento
        Si tipoCliente = 1 Entonces
            descuento <- montoVenta * 0.30
        Sino
            descuento <- 0
        FinSi
		
        // Calcular subtotal
        subTotal <- montoVenta - descuento
		
        // Calcular impuesto
        impuesto <- subTotal * 0.13
		
        // Calcular total a pagar
        totalPagar <- subTotal + impuesto
		
        // Imprimir factura
        Escribir "============================="
        Escribir "         FACTURA #" , numeroFactura
        Escribir "============================="
        Escribir "Cedula: " , cedula
        Escribir "Nombre: " , nombre
        Escribir "Tipo de cliente: " , tipoCliente
        Escribir "Tipo de tiquete: " , tipoTiquete
        Escribir "Cantidad de tiquetes: " , cantidadTiquetes
        Escribir "Monto unitario por tiquete: " , montoUnitario
        Escribir "Monto de venta: " , montoVenta
        Escribir "Descuento: " , descuento
        Escribir "Subtotal de venta: " , subTotal
        Escribir "Impuesto de venta: " , impuesto
        Escribir "Total a pagar: " , totalPagar
        Escribir "============================="
		
        // Actualizar estadísticas
        Si tipoTiquete = 1 Entonces
            cantGaleria <- cantGaleria + cantidadTiquetes
            acumGaleria <- acumGaleria + totalPagar
        Sino
            cantPalco <- cantPalco + cantidadTiquetes
            acumPalco <- acumPalco + totalPagar
        FinSi
        cantTotal <- cantTotal + cantidadTiquetes
        acumTotal <- acumTotal + totalPagar
		
        Escribir "Desea registrar otra venta? (s/n): "
        Leer continuar
		
    FinMientras
	
    // Mostrar estadísticas finales
    Escribir "Estadisticas finales:"
    Escribir "============================="
    Escribir "Cantidad de entradas Tiquetes Galeria: " , cantGaleria
    Escribir "Acumulado Dinero por Tiquetes Galeria: " , acumGaleria
    Escribir "Cantidad de entradas Tiquetes Palco: " , cantPalco
    Escribir "Acumulado Dinero por Tiquetes Palco: " , acumPalco
    Escribir "Cantidad General de Entradas: " , cantTotal
    Escribir "Acumulado General Dinero por Entradas: " , acumTotal
    Si numeroFactura > 0 Entonces
        Escribir "Promedio General por Ventas: " , (acumTotal / numeroFactura)
    FinSi
    Escribir "============================="
	
FinProceso
